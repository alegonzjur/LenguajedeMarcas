/* Ejercicio 1*/
SELECT nomde,
nomce, 
max(salar)
FROM EMPLEADOS E JOIN DEPARTAMENTOS D
    ON E.numde = D.numde 
    JOIN CENTROS C 
    ON D.numce = C.numce
WHERE presu < 35
GROUP BY D.nomde, nomce
HAVING MAX(salar) >1500
ORDER BY numce ASC;

/*Ejercicio 4*/
SELECT nomde,
COUNT(numem),
SUM(salar),
SUM(comis),
SUM(numhi)
FROM DEPARTAMENTOS D JOIN EMPLEADOS E 
ON D.numde = E.numde 
WHERE tidir = 'F'
GROUP by nomde;

/*Ejercicio 5*/

extel,
COUNT(numem),
FROM DEPARTAMENTOS D JOIN EMPLEADOS E
ON D.numde = E.numde
WHERE presu > 35
GROUP BY nomde, extel;
