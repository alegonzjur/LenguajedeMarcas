CREATE TABLE EMPRESAS(
    CIF INTEGER PRIMARY KEY,
    Nombre VARCHAR2(40),
    Direccion VARCHAR2(40),
    Telefono INTEGER
);

CREATE TABLE ALUMNOS(
    DNI_Alu CHAR(9) PRIMARY KEY,
    Nombre VARCHAR2(40),
    Apellido1 VARCHAR2(40),
    Apellido2 VARCHAR2(40),
    Telefono INTEGER,
    Edad INTEGER,
    CIF INTEGER
);

CREATE TABLE asistir(
    N_Curso INTEGER,
    DNI_Alu CHAR(9),
    Nota NUMBER(4)
);  

CREATE TABLE CURSOS(
    N_Curso INTEGER PRIMARY KEY,
    Fecha_Inicio DATE,
    Fecha_Fin DATE,
    DNI_Prof CHAR(9),
    Cod_TipCurso INTEGER
);

CREATE TABLE PROFESORES(
    DNI_Prof CHAR(9) PRIMARY KEY,
    Apellido1 VARCHAR2(40),
    Apellido2 VARCHAR2(40),
    Direccion VARCHAR2(40),
    Telefono INTEGER
);

CREATE TABLE TIPOS_CURSO(
    Cod_TipCurso INTEGER PRIMARY KEY,
    Titulo VARCHAR2(40),
    Duracion VARCHAR2(20)
);

ALTER TABLE ALUMNOS ADD(
    CONSTRAINT FK_ALU FOREIGN KEY (CIF) REFERENCES EMPRESAS(CIF)
);
ALTER TABLE asistir ADD(
    CONSTRAINT PK_asi PRIMARY KEY (N_Curso,DNI_Alu),
    CONSTRAINT FK1_asi FOREIGN KEY (N_Curso) REFERENCES CURSOS(N_Curso),
    CONSTRAINT FK2_asi FOREIGN KEY (DNI_Alu) REFERENCES ALUMNOS(DNI_Alu)
);
ALTER TABLE CURSOS ADD(
    CONSTRAINT FK1_CUR FOREIGN KEY (DNI_Prof) REFERENCES PROFESORES(DNI_Prof),
    CONSTRAINT FK2_CUR FOREIGN KEY (Cod_TipCurso) REFERENCES TIPOS_CURSO(Cod_TipCurso)
);
