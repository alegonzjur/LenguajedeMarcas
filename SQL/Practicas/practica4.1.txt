CREATE TABLE CENTROS(
    numce NUMBER(4) PRIMARY KEY NOT NULL,
    nomce VARCHAR2(25),
    dirce VARCHAR2(25)
);

CREATE TABLE DEPARTAMENTOS(
    numde NUMBER(3) PRIMARY KEY NOT NULL,
    numce NUMBER(4),
    direc NUMBER(3),
    tidir CHAR(1),
    presu NUMBER(3,1),
    depde NUMBER(3),
    nomde VARCHAR2(20)
);

CREATE TABLE EMPLEADOS(
    nummem INTEGER PRIMARY KEY NOT NULL,
    extel NUMBER(3),
    fecna DATE,
    fecin DATE,
    salar NUMBER(5),
    comis NUMBER(3),
    numhi NUMBER(1),
    nomem VARCHAR2(10),
    numde NUMBER(3)
);

ALTER TABLE DEPARTAMENTOS ADD(
    CONSTRAINT FK_DEP1 FOREIGN KEY (numce) REFERENCES CENTROS(numce),
    CONSTRAINT FK_DEP2 FOREIGN KEY (depde) REFERENCES DEPARTAMENTOS(numde)
);
